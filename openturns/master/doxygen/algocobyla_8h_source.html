44 /*
45  * Verbosity level
46  */
47 typedef enum {
48  COBYLA_MSG_NONE = 0, /* No messages */
49  COBYLA_MSG_EXIT = 1, /* Exit reasons */
50  COBYLA_MSG_ITER = 2, /* Rho and Sigma changes */
51  COBYLA_MSG_INFO = 3, /* Informational messages */
52 } cobyla_message;

54 /*
55  * Possible return values for cobyla
56  */
57 typedef enum
58  {
59  COBYLA_MINRC = -2, /* Constant to add to get the rc_string */
60  COBYLA_EINVAL = -2, /* N<0 or M<0 */
61  COBYLA_ENOMEM = -1, /* Memory allocation failed */
62  COBYLA_NORMAL = 0, /* Normal return from cobyla */
63  COBYLA_MAXFUN = 1, /* Maximum number of function evaluations reach */
64  COBYLA_ROUNDING = 2, /* Rounding errors are becoming damaging */
65  COBYLA_USERABORT = 3 /* User requested end of minimization */
66  } cobyla_rc;

68 /*
69  * Return code strings
70  * use cobyla_rc_string[rc - COBYLA_MINRC] to get the message associated with
71  * return code rc.
72  */
73 extern char *cobyla_rc_string[6];

75 /*
76  * A function as required by cobyla
77  * state is a void pointer provided to the function at each call
78  *
79  * n : the number of variables
80  * m : the number of constraints
81  * x : on input, then vector of variables (should not be modified)
82  * f : on output, the value of the function
83  * con : on output, the value of the constraints (vector of size m)
84  * state : on input, the value of the state variable as provided to cobyla
85  *
86  * COBYLA will try to make all the values of the constraints positive.
87  * So if you want to input a constraint j such as x[i] <= MAX, set:
88  * con[j] = MAX - x[i]
89  * The function must returns 0 if no error occurs or 1 to immediately end the
90  * minimization.
91  *
92  */
93 typedef int cobyla_function(int n, int m, double *x, double *f, double *con,
94  void *state);
95 
96 /*
97  * cobyla : minimize a function subject to constraints
98  *
99  * n : number of variables (>=0)
100  * m : number of constraints (>=0)
101  * x : on input, initial estimate ; on output, the solution
102  * rhobeg : a reasonable initial change to the variables
103  * rhoend : the required accuracy for the variables
104  * message : see the cobyla_message enum
105  * maxfun : on input, the maximum number of function evaluations
106  * on output, the number of function evaluations done
107  * calcfc : the function to minimize (see cobyla_function)
108  * state : used by function (see cobyla_function)
109  *
110  * The cobyla function returns a code defined in the cobyla_rc enum.
111  *
112  */
113 extern int ot_cobyla(int n, int m, double *x, double rhobeg, double rhoend,
114  int message, int *maxfun, cobyla_function *calcfc, void *state);
115 
116 END_C_DECLS
