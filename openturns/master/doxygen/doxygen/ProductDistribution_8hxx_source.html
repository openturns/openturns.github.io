175  // Structure used to wrap the kernel of the integral defining the PDF of the product
176  struct PDFKernelWrapper
177  {
178  const Distribution left_;
179  const Distribution right_;
180  const Scalar x_;
181  const Bool isZero_;
182  const Scalar pdf0_;
183 
184  PDFKernelWrapper(const Distribution & left,
185  const Distribution & right,
186  const Scalar x):
187  left_(left), right_(right), x_(x), isZero_(std::abs(x) < ResourceMap::GetAsScalar("Distribution-DefaultQuantileEpsilon")), pdf0_(isZero_ ? right.computePDF(0.0) : 0.0) {};
188 
189  Point eval(const Point & point) const
190  {
191  const Scalar value = left_.computePDF(point);
192  if (value == 0.0) return Point(1, 0.0);
193  const Scalar u = point[0];
194  const Scalar absU = std::abs(u);
195  // x_ == 0
196  if (isZero_)
197  {
198  if (pdf0_ == 0.0) return Point(1, 0.0);
199  if (absU == 0.0) return Point(1, SpecFunc::MaxScalar);
200  return Point(1, value * pdf0_ / absU);
201  }
202  // x_ != 0
203  if (absU == 0.0)
204  {
205  const Scalar epsilon = 1e-7;
206  return Point(1, value * 0.5 * (right_.computePDF(x_ / epsilon) + right_.computePDF(-x_ / epsilon)) / epsilon);
207  }
208  return Point(1, value * right_.computePDF(x_ / u) / absU);
209  };
210  }; // struct PDFKernelWrapper
