34 /*
35  * class GenericCounterImplementation
36  *
37  * brief This class is in charge of the reference counter for the shared pointer.
38  *
39  * This class implements the base mechanisms of reference counting.
40  * It keeps track of the number of references to an hypothetic pointee (not defined here
41  * because the class is abstract).
42  *
43  * This class, its interface and its implementation are rewrittings of the BOOST shared_ptr
44  * template class. We forgived using BOOST because of some pain in compiling OpenTURNS on
45  * some architectures (with the help of Debian autobuilder) but the implementation was
46  * so smart we decided to reimplement ours (with lesser dependencies) by cloning the interface
47  * and part of the implementation of the BOOST one. Many thanks to BOOST for its work.
48  */
49 
50 class OT_API GenericCounterImplementation
51 {
52  /* The number of references to the pointee */
53  AtomicInt use_count_;
54 
55 public:
56 
57  /*
58  * brief Default constructor.
59  *
60  * It initializes the reference counter to one.
61  */
62  GenericCounterImplementation() : use_count_()
63  {
64  use_count_.increment();
65  }
66 
67  /* Destructor */
68  virtual ~GenericCounterImplementation() {}
69 
70  /*
71  * brief Delete the pointee.
72  *
73  * This method diposes of the pointee by deleting it because it isn't referenced any more.
74  * This method is pure because the allocation and deallocation are not handled here. The
75  * derived class is in charge ot this functionnality.
76  * See also: release()
77  */
78  virtual void dispose() = 0;
79 
80  /*
81  * brief Remove one reference.
82  *
83  * This method removes one reference to the counter. If it is the last one, it calls
84  * dispose() to delete the pointee.
85  * See also: dispose() and add_ref_copy()
86  */
87  void release()
88  {
89  if ( use_count_.fetchAndAdd( -1 ) == 1 ) dispose();
90  }
91 
92  /*
93  * brief Add one reference.
94  *
95  * This method adds one reference to the counter.
96  * See also: release()
97  */
98  void add_ref_copy()
99  {
100  use_count_.increment();
101  }
102 
103  /*
104  * brief Get the number of references.
105  *
106  * This methods returns the number of references currently owned by the counter.
107  * Return: the number of references
108  * See also: release() and add_ref_copy()
109  */
110  UnsignedInteger use_count()
111  {
112  return use_count_.get();
113  }
