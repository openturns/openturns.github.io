226  inline
227  const ValueType find(const KeyType & key) const
228  {
229  if (!enabled_) return ValueType();
230 
231  typename map_type::iterator it = points_.find(key);
232  Bool found = (it != points_.end());
233  if (!found) return ValueType();
234 
235  list_key_iterator pos = it->second.second;
236  if (pos != keys_.begin())
237  {
238  // Move key at the beginning
239  keys_.erase(pos);
240  keys_.push_front(key);
241  it->second.second = keys_.begin();
242  }
243  LOGDEBUG(OSS() << "Cache hit !");
244  ++hits_;
245  return it->second.first;
246  }

249  inline
250  void add(const KeyType & key,
251  const ValueType & value)
252  {
253  // Do not allow the key to be added twice
254  if (!enabled_ || (points_.find(key) != points_.end())) return;
255 
256  if (points_.size() == maxSize_)
257  {
258  // Remove the last element
259  list_key_iterator last = keys_.end();
260  --last;
261  points_.erase(*last);
262  keys_.erase(last);
263  }
264 
265  // Insert the element at the beginning
266  keys_.push_front(key);
267 
268  points_[key] = std::make_pair(value, keys_.begin());
269  }
