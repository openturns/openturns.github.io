89  // It is a simple call to the computeReducedLogLikelihood() of the algo
90  Point operator()(const Point &point) const override
91  {
92  const Scalar gamma = point[0];
93  const Point sums(computeMaximumLikelihoodSums(gamma));
94  return Point(1, sums[0] * (sums[2] - sums[1] * (1.0 + sums[1] / size_)) + size_ * sums[3]);
95  }
96 
97  /*
98  S_0 = \sum_i (X_i - \gamma)^{-1}
99  S_1 = \sum_i \log(X_i - \gamma)
100  S_2 = \sum_i \log^2(X_i - \gamma)
101  S_3 = \sum_i \log(X_i - \gamma) / (X_i - \gamma)
102  */
103  Point computeMaximumLikelihoodSums(const Scalar gamma) const
104  {
105  Point sums(4, 0.0);
106  for (UnsignedInteger i = 0; i < size_; ++i)
107  {
108  const Scalar delta = sample_(i, 0) - gamma;
109  if (!(delta > 0.0))
110  throw InvalidArgumentException(HERE) << "Error: cannot estimate a LogNormal distribution based on the given sample using the method of local maximum likelihood, probably because the sample is constant.";
111  const Scalar logDelta = std::log(delta);
112  const Scalar inverseDelta = 1.0 / delta;
113  sums[0] += inverseDelta;
114  sums[1] += logDelta;
115  sums[2] += logDelta * logDelta;
116  sums[3] += logDelta * inverseDelta;
117  }
118  return sums;
119  }
