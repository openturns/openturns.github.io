224  const ValueType find(const KeyType & key) const
225  {
226  if (!enabled_) return ValueType();
227 
228  typename map_type::iterator it = points_.find(key);
229  Bool found = (it != points_.end());
230  if (!found) return ValueType();
231 
232  list_key_iterator pos = it->second.second;
233  if (pos != keys_.begin())
234  {
235  // Move key at the beginning
236  keys_.erase(pos);
237  keys_.push_front(key);
238  it->second.second = keys_.begin();
239  }
240  LOGDEBUG(OSS() << "Cache hit !");
241  ++hits_;
242  return it->second.first;
243  }

246  inline
247  void add(const KeyType & key,
248  const ValueType & value)
249  {
250  // Do not allow the key to be added twice
251  if (!enabled_ || (points_.find(key) != points_.end())) return;
252 
253  if (points_.size() == maxSize_)
254  {
255  // Remove the last element
256  list_key_iterator last = keys_.end();
257  --last;
258  points_.erase(*last);
259  keys_.erase(last);
260  }
261 
262  // Insert the element at the beginning
263  keys_.push_front(key);
264 
265  points_[key] = std::make_pair(value, keys_.begin());
266  }
