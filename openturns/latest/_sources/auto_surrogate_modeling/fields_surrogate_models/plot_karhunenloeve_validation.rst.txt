    numberOfVertices = 20
    interval = ot.Interval(-1.0, 1.0)
    mesh = ot.IntervalMesher([numberOfVertices - 1]).build(interval)
    covariance = ot.SquaredExponential()
    process = ot.GaussianProcess(covariance, mesh)
